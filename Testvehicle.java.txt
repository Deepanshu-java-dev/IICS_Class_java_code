interface Vehicle
{
	int speed
	void start();
	void stop();
	int getSpeed();
}
class bike extend Vehicle
{	
	int speed
 	public void start()
	{
		System.out.println("Bike Started");
		speed=10;
	}
	public void stop();
	{
	 System.out.println("Bike Stopped");
		speed=0;
	}
	int getSpeed()
	{
		return speed;
	}
}
